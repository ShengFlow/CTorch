28uint32_t computeCode(ErrorLevel level, ErrorPlatform platform, ErrorType type) {
29 uint32_t code = 0;
30 code |= (static_cast<uint32_t>(platform) << 24); // 平台：高8位
31 code |= (static_cast<uint32_t>(type) << 16); // 类型：次8位
32 code |= (static_cast<uint32_t>(level) << 8); // 级别：次8位
33 // 最后8位保留（可用于扩展具体场景码）
34 return code;
35}

41std::string getFormattedTimeMs() {
42 using namespace std::chrono;
43
44 // 获取当前时间（精度：ms）
45 const auto now = system_clock::now();
46 const auto now_ms = time_point_cast<milliseconds>(now);
47 const uint64_t ms = now_ms.time_since_epoch().count() % 1000; // 提取毫秒位（0-999）
48
49 // 转换为秒级time_t，用于格式化日期时间
50 const std::time_t now_t = system_clock::to_time_t(now);
51
52 // 线程安全的时间解析（避免localtime线程不安全问题）
53 std::tm tm_buf{};
54#ifdef _WIN32
55 localtime_s(&tm_buf, &now_t); // Windows线程安全版本
56#else
57 localtime_r(&now_t, &tm_buf); // Linux/macOS线程安全版本
58#endif
59
60 // 4. 格式化字符串（补零到3位毫秒）
61 std::ostringstream oss;
62 oss << std::put_time(&tm_buf, "%Y-%m-%d %H:%M:%S")
63 << "." << std::setw(3) << std::setfill('0') << ms;
64
65 return oss.str();
66}
