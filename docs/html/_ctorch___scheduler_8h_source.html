98// 公共接口实现：dispatch（双输入算子）
99 Tensor dispatch(const Tensor& a, const Tensor& b, op op_type) {
100 // 1. 参数合法性校验（形状、dtype一致）
101 if (a.sizes() != b.sizes()) {
102 Ctorch_Error::log(ErrorLevel::ERROR,ErrorPlatform::kGENERAL,ErrorType::DIMENSION,"Ctorch_Scheduler: Tensor形状不一致");
103 }
104 if (a.dtype() != b.dtype()) {
105 Ctorch_Error::log(ErrorLevel::ERROR,ErrorPlatform::kGENERAL,ErrorType::DATATYPE,"Ctorch_Scheduler: Tensor类型不一致");
106 }
107
108 // 获取调度器实例，初始化kernel映射表（仅首次调用）
109 Ctorch_Scheduler &instance = getInstance();
110 std::lock_guard<std::mutex> lock(instance.mutex_);
111 static bool kernel_map_inited = false;
112 if (!kernel_map_inited) {
113 instance.initKernelMap();
114 kernel_map_inited = true;
115 }
116
117 // 确定目标设备，查找可用kernel
118 DeviceType target_dev = getTargetDevice(a, b);
119 BinaryKernelFunc target_kernel = nullptr;
120
121 // 从映射表中查找对应kernel
122 auto op_it = instance.binary_kernel_map_.find(op_type);
123 if (op_it != instance.binary_kernel_map_.end()) {
124 auto dev_it = op_it->second.find(target_dev);
125 if (dev_it != op_it->second.end() && isDeviceAvailable(target_dev)) {
126 target_kernel = dev_it->second;
127 }
128 }
129
130 // 未找到kernel则抛异常
131 if (target_kernel == nullptr) {
132 Ctorch_Error::log(ErrorLevel::ERROR,ErrorPlatform::kGENERAL,ErrorType::PLATFORM_API,"Ctorch_Scheduler: 没有可用的Kernel");
133 }
134 // 调用kernel，执行计算并返回结果
135 return target_kernel(a, b);
136 }
