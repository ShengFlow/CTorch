92void Tensor::computeStrides() {
93 _strides.resize(_shape.size());
94 if (_shape.empty()) {
95 return;
96 }
97 _strides.back() = 1;
98 for (int i = static_cast<int>(_shape.size()) - 2; i >= 0; --i) {
99 _strides[i] = _strides[i + 1] * _shape[i + 1];
100 }
101}

109size_t Tensor::computeStorageIndex(std::initializer_list<size_t> indices) const {
110 if (indices.size() != _shape.size()) {
111 Ctorch_Error::throwException(ErrorPlatform::kGENERAL, ErrorType::DIMENSION, "索引维度与张量维度不匹配");
112 }
113 size_t index = 0;
114 auto indices_it = indices.begin();
115 auto strides_it = _strides.begin();
116 for (; indices_it != indices.end() && strides_it != _strides.end(); ++indices_it, ++strides_it) {
117 index += *indices_it * *strides_it;
118 }
119 return index + _storage_offset;
120}

614// 全局的matMul函数
615Tensor matMul(const Tensor& a, const Tensor& b) {
616 // 简单实现，仅支持2D张量
617 if (a.shape().size() != 2 || b.shape().size() != 2) {
618 throw std::invalid_argument("matMul仅支持2D张量");
619 }
620 if (a.shape()[1] != b.shape()[0]) {
621 throw std::invalid_argument("矩阵维度不匹配");
622 }
623
624 size_t m = a.shape()[0];
625 size_t k = a.shape()[1];
626 size_t n = b.shape()[1];
627
628 Tensor result(ShapeTag{}, {m, n}, a.dtype(), a.device());
629 // 简单的矩阵乘法实现
630 for (size_t i = 0; i < m; ++i) {
631 for (size_t j = 0; j < n; ++j) {
632 float sum = 0.0f;
633 for (size_t l = 0; l < k; ++l) {
634 sum += a.data<float>()[i * k + l] * b.data<float>()[l * n + j];
635 }
636 result.data<float>()[i * n + j] = sum;
637 }
638 }
639
640 return result;
641}
