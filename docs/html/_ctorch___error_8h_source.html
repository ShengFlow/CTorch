353 static void log(ErrorLevel level, ErrorPlatform platform, ErrorType type, const std::string msg) {
354 Ctorch_Stats& inst = Ctorch_Stats::getInstance();
355 if (inst.level == PrintLevel::MEDIUM && level == ErrorLevel::TRACE) return;
356 if (inst.level == PrintLevel::MINIUM && (level == ErrorLevel::DEBUG || level == ErrorLevel::TRACE)) return ;
357 uint32_t error_code = computeCode(level, platform, type);
358 switch (level) {
359 case ErrorLevel::TRACE: {
360 printf(ESC_START COLOR_TRACE);
361 break;
362 }
363 case ErrorLevel::INFO: {
364 printf(ESC_START COLOR_INFO);
365 break;
366 }
367 case ErrorLevel::DEBUG: {
368 printf(ESC_START COLOR_DEBUG);
369 break;
370 }
371 case ErrorLevel::WARN: {
372 printf(ESC_START COLOR_WARN);
373 break;
374 }
375 case ErrorLevel::ERROR: {
376 printf(ESC_START COLOR_ERR);
377 break;

379 case ErrorLevel::FATAL: {
380 printf(ESC_START COLOR_FATAL);

389 printf("[%s] [%s %" PRIu64 "] [ERROR_CODE:0x%" PRIX32 "] [PLATFORM:%s] [TYPE:%s] %s\n",

391 getFormattedTimeMs().c_str(),
392 getTimestampMs(),
393 error_code,
394 getPlatformName(platform).c_str(),
395 getTypeName(type).c_str(),
396 msg.c_str());
397 printf(ESC_END);
398 if (level == ErrorLevel::ERROR) {
399 inst.incrError();
400 } else if (level == ErrorLevel::WARN) {
401 inst.incrWarn();

403 inst.incrFatal();
