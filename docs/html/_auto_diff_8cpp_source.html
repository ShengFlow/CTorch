85Tensor AutoDiff::get_grad(const Tensor* t) {
86 if (!t || t->id() == 0) {
87 Ctorch_Error::trace(ErrorPlatform::kCPU, ">>> get_grad: 无效输入");
88 return Tensor();
89 }
90
91 std::ostringstream oss;
92 oss << ">>> get_grad - 开始, 目标ID: " << t->id();
93 std::string msg = oss.str();
94 Ctorch_Error::trace(ErrorPlatform::kCPU, msg);
95
96 // 保存输入张量的信息，用于创建默认零张量
97 std::vector<size_t> input_shape = t->shape();
98 DType input_dtype = t->dtype();
99 DeviceType input_device = t->device();
100
101 // 第一步：在锁内获取必要信息（不创建新Tensor）
102 bool has_grad = false;
103 std::vector<size_t> grad_shape;
104 DType grad_dtype;
105 DeviceType grad_device;
106 float grad_value = 0.0f; // 存储梯度值用于调试
107
108 {
109 std::lock_guard<std::mutex> lock(records_mutex);
110 auto it = id_to_node.find(t->id());
111 if (it != id_to_node.end() && it->second && it->second->requires_grad && it->second->grad) {
112 has_grad = true;
113 grad_shape = it->second->grad->sizes();
114 grad_dtype = it->second->grad->dtype();
115 grad_device = it->second->grad->device();
116
117 // 获取梯度值用于调试
118 if (grad_dtype == DType::kFloat && !grad_shape.empty()) {
119 const float *grad_data = it->second->grad->data<float>();
120 if (grad_data) {
121 grad_value = grad_data[0]; // 对于标量，取第一个元素
122 }
123 }
124 std::ostringstream osss;
125 osss << ">>> 找到梯度，形状: [";
126 for (auto s: grad_shape) osss << s << " ";
127 osss << "], 值: " << grad_value;
128 std::string msgs = osss.str();
129 Ctorch_Error::trace(ErrorPlatform::kCPU, msgs);
130 }
131 } // 释放锁
132
133 // 第二步：在锁外创建结果Tensor
134 if (has_grad) {
135 Ctorch_Error::trace(ErrorPlatform::kCPU, ">>> 创建梯度副本");
136 Tensor result(ShapeTag{}, grad_shape, grad_dtype, grad_device);
137
138 // 第三步：重新加锁复制数据
139 {
140 std::lock_guard<std::mutex> lock(records_mutex);
141 auto it = id_to_node.find(t->id());
142 if (it != id_to_node.end() && it->second && it->second->grad) {
143 Ctorch_Error::trace(ErrorPlatform::kCPU, ">>> 复制梯度数据");
144 // 直接复制数据，避免调用clone()
145 switch (grad_dtype) {
146 case DType::kFloat: {
147 const float *src = it->second->grad->data<float>();
148 float *dst = result.data<float>();
149 size_t count = result.numel();
150 for (size_t i = 0; i < count; ++i) {
151 dst[i] = src[i];
152 }
153 std::ostringstream osss;
154 osss << ">>> 复制了 " << count << " 个float值，第一个值: " << dst[0];
155 std::string msgs = osss.str();
156 Ctorch_Error::trace(ErrorPlatform::kCPU, msgs);
157 break;
158 }
159 case DType::kDouble: {
160 const double *src = it->second->grad->data<double>();
161 double *dst = result.data<double>();
162 for (size_t i = 0; i < result.numel(); ++i) {
163 dst[i] = src[i];
164 }
165 break;
166 }
167 case DType::kInt: {
168 const int32_t *src = it->second->grad->data<int32_t>();
169 int32_t *dst = result.data<int32_t>();
170 for (size_t i = 0; i < result.numel(); ++i) {
171 dst[i] = src[i];
172 }
173 break;
174 }
175 case DType::kLong: {
176 const int64_t *src = it->second->grad->data<int64_t>();
177 int64_t *dst = result.data<int64_t>();
178 for (size_t i = 0; i < result.numel(); ++i) {
179 dst[i] = src[i];
180 }
181 break;
182 }
183 default:
184 Ctorch_Error::throwException(ErrorPlatform::kCPU, ErrorType::DATATYPE, "get_grad中不支持的dtype");
185 }
186 }
187 }
188 std::ostringstream osss;
189 osss << ">>> 最终梯度副本值: " << result.item<float>() << std::endl;
190 osss << "<<< get_grad - 完成" << std::endl;
191
192 std::string msgs = osss.str();
193 Ctorch_Error::trace(ErrorPlatform::kCPU, msgs);
194 return result;
195 }
196 // 未找到梯度时，输出Error级别提示并返回零张量
197 Ctorch_Error::trace(ErrorPlatform::kCPU, "[ERROR] [2025-12-21 18:05:10.882 1766311510882] [ERROR_CODE:0x5060400] [PLATFORM:GENERAL] [TYPE:TENSOR_STATE] Tensor梯度未找到，返回零张量");
198 // 创建与输入张量相同形状、数据类型和设备的零张量
199 Tensor result(ShapeTag{}, input_shape, input_dtype, input_device);
200 // 初始化所有元素为0
201 switch (input_dtype) {
202 case DType::kFloat: {
203 float *dst = result.data<float>();
204 size_t count = result.numel();
205 for (size_t i = 0; i < count; ++i) {
206 dst[i] = 0.0f;
207 }
208 break;
209 }
210 case DType::kDouble: {
211 double *dst = result.data<double>();
212 size_t count = result.numel();
213 for (size_t i = 0; i < count; ++i) {
214 dst[i] = 0.0;
215 }
216 break;
217 }
218 case DType::kInt: {
219 int32_t *dst = result.data<int32_t>();
220 size_t count = result.numel();
221 for (size_t i = 0; i < count; ++i) {
222 dst[i] = 0;
223 }
224 break;
225 }
226 case DType::kLong: {
227 int64_t *dst = result.data<int64_t>();
228 size_t count = result.numel();
229 for (size_t i = 0; i < count; ++i) {
230 dst[i] = 0;
231 }
232 break;
233 }
234 default:
235 Ctorch_Error::throwException(ErrorPlatform::kCPU, ErrorType::DATATYPE, "get_grad中不支持的dtype");
236 }
237 return result;
238}

291void AutoDiff::defer_record(size_t output_id, op operation, const std::vector<Tensor*>& inputs) {
292 std::ostringstream oss;
293 oss << ">>> 进入 defer_record, output_id: " << output_id;
294 std::string msg = oss.str();
295 Ctorch_Error::trace(ErrorPlatform::kCPU, msg);
296 if (output_id == 0) {
297 Ctorch_Error::log(ErrorLevel::WARN, ErrorPlatform::kCPU, ErrorType::TENSOR_STATE, "警告: output_id 为0");
298 return;
299 }
300
301 // 创建记录对象（不持有锁）
302 PendingRecord record;
303 record.operation = operation;
304
305 std::vector<size_t> input_ids;
306 std::vector<std::vector<size_t>> input_shapes;
307
308 // 收集输入信息（不持有锁）
309 for (Tensor* input : inputs) {
310 if (input && input->id() != 0) {
311 input_ids.push_back(input->id());
312 input_shapes.push_back(input->shape());
313 std::ostringstream osss;
314 osss << ">>> 处理输入: " << input->id();
315 std::string msgs = osss.str();
316 Ctorch_Error::trace(ErrorPlatform::kCPU, msgs);
317
318 // 检查是否已注册，避免不必要的锁
319 bool needs_registration = false;
320 {
321 std::lock_guard<std::mutex> lock(records_mutex);
322 needs_registration = (id_to_node.find(input->id()) == id_to_node.end());
323 }
324
325 if (needs_registration) {
326 std::ostringstream ost;
327 ost << ">>> 注册叶子节点: " << input->id() << std::endl;
328 std::string msgt = ost.str();
329 Ctorch_Error::trace(ErrorPlatform::kCPU, msgt);
330
331 make_leaf(*input, input->requires_grad());
332 }
333 }
334 }
335
336 // 现在持有锁并设置记录
337 {
338 std::lock_guard<std::mutex> lock(records_mutex);
339 record.input_ids = input_ids;
340 record.input_shapes = input_shapes;
341 pending_records[output_id] = record;
342 }
343
344 Ctorch_Error::trace(ErrorPlatform::kCPU, "<<< 离开 defer_record");
345}

347// ======================= commit_record函数 =======================
348void AutoDiff::commit_record(Tensor& output) {
349 size_t output_id = output.id();
350 std::cout << ">>> AutoDiff::commit_record - 开始, ID: " << output_id << std::endl;
351
352 if (output_id == 0) {
353 std::cout << "!!! 错误: 输出ID为0" << std::endl;
354 return;
355 }
356
357 debug_print_state("commit_record开始前");
358
359 // 第一步：收集必要信息（持有锁，但时间很短）
360 PendingRecord record_copy;
361 std::vector<size_t> input_ids_copy;
362 bool should_create_node = false;
363
364 {
365 std::lock_guard<std::mutex> lock(records_mutex);
366
367 auto it = pending_records.find(output_id);
368 if (it == pending_records.end()) {
369 std::cout << "!!! 警告: 找不到待处理记录 " << output_id << std::endl;
370 return;
371 }
372
373 PendingRecord& record = it->second;
374 if (record.committed) {
375 std::cout << ">>> 记录 " << output_id << " 已提交，跳过" << std::endl;
376 return;
377 }
378
379 std::cout << ">>> 开始提交记录 " << output_id << std::endl;
380 std::cout << ">>> 操作类型: " << static_cast<int>(record.operation) << std::endl;
381 std::cout << ">>> 输入IDs: ";
382 for (auto id : record.input_ids) std::cout << id << " ";
383 std::cout << std::endl;
384
385 // 验证输入节点
386 bool valid = true;
387 for (size_t input_id : record.input_ids) {
388 if (id_to_node.find(input_id) == id_to_node.end()) {
389 std::cout << "!!! 错误: 输入节点 " << input_id << " 不存在" << std::endl;
390 valid = false;
391 break;
392 } else {
393 std::cout << ">>> 输入节点 " << input_id << " 存在" << std::endl;
394 }
395 }
396
397 if (!valid) {
398 std::cout << "!!! 记录验证失败，清除记录 " << output_id << std::endl;
399 pending_records.erase(it);
400 return;
401 }
402
403 // 复制必要信息，然后释放锁
404 record_copy = record;
405 input_ids_copy = record.input_ids;
406 should_create_node = true;
407
408 // 立即标记为已提交，避免重复处理
409 record.committed = true;
410 } // 释放锁
411
412 // 第二步：在锁外创建节点（避免死锁）
413 if (should_create_node) {
414 std::cout << ">>> 在锁外创建操作节点 " << output_id << std::endl;
415
416 // 确定梯度需求（在锁外检查）
417 bool requires_grad = false;
418 for (size_t input_id : input_ids_copy) {
419 Node* input_node = get_node_by_id(input_id);
420 if (input_node && input_node->requires_grad) {
421 requires_grad = true;
422 break;
423 }
424 }
425
426 std::cout << ">>> 节点 " << output_id << " 需要梯度: " << requires_grad << std::endl;
427
428 // 创建节点（不持有锁）
429 auto tensor_ptr = std::make_unique<Tensor>(output.clone());
430 auto output_node = std::make_unique<Node>(output_id, std::move(tensor_ptr), requires_grad, false);
431 output_node->operation = record_copy.operation;
432 output_node->input_ids = input_ids_copy;
433
434 // 延迟创建梯度存储
435 if (requires_grad) {
436 std::cout << ">>> 为节点 " << output_id << " 延迟分配梯度存储" << std::endl;
437 // 注意：不在构造函数中创建 grad，避免死锁
438 }
439
440 // 第三步：重新加锁，快速插入节点
441 {
442 std::lock_guard<std::mutex> lock(records_mutex);
443 id_to_node[output_id] = std::move(output_node);
444
445 // 清理已提交的记录
446 auto it = pending_records.find(output_id);
447 if (it != pending_records.end() && it->second.committed) {
448 pending_records.erase(it);
449 }
450 }
451
452 std::cout << "<<< 记录提交完成 " << output_id << std::endl;
453 debug_print_state("commit_record完成后");
454 }
455}

498// ======================= 私有辅助函数：DFS拓扑排序 =======================
505void AutoDiff::dfs_topological_sort(Node* node, std::unordered_set<Node*>& visited, std::vector<Node*>& result) {
506 if (!node) return;
507 if (visited.find(node) != visited.end()) return;
508
509 visited.insert(node);
510
511 // 先访问所有输入节点（依赖节点）
512 for (size_t input_id : node->input_ids) {
513 auto input_it = id_to_node.find(input_id);
514 if (input_it != id_to_node.end() && input_it->second) {
515 dfs_topological_sort(input_it->second.get(), visited, result);
516 }
517 }
518
519 // 后序遍历：当前节点处理完所有依赖后，加入结果
520 result.push_back(node);
521}

523// ======================= backward函数（版本2） =======================
524void AutoDiff::backward(Tensor& root, Tensor grad_output) {
525 std::cout << ">>> =========================================" << std::endl;
526 std::cout << ">>> 进入 backward 函数，root ID: " << root.id() << std::endl;
527
528 if (root.id() == 0) {
529 Ctorch_Error::throwException(ErrorPlatform::kCPU, ErrorType::TENSOR_STATE, "反向传播的无效根张量 (ID为0)");
530 return;
531 }
532
533 // 阶段1：准备阶段（不持有锁）
534 std::cout << ">>> 阶段1: 准备阶段" << std::endl;
535
536 // 阶段2：验证阶段和设置根节点梯度
537 std::cout << ">>> 阶段2: 验证和设置根节点梯度" << std::endl;
538 Node* root_node = nullptr;
539 bool root_requires_grad = false;
540 std::vector<size_t> root_shape;
541
542 {
543 std::lock_guard<std::mutex> lock(records_mutex);
544 auto it = id_to_node.find(root.id());
545 if (it == id_to_node.end() || !it->second) {
546 std::cout << "!!! 错误: 找不到根节点 " << root.id() << std::endl;
547 return;
548 }
549 root_node = it->second.get();
550 root_requires_grad = root_node->requires_grad;
551 root_shape = root.sizes();
552
553 if (!root_requires_grad) {
554 std::cout << ">>> 根节点不需要梯度，跳过反向传播" << std::endl;
555 return;
556 }
557
558 // 设置根节点梯度
559 if (!root_node->grad) {
560 root_node->grad = std::make_unique<Tensor>(ShapeTag{}, root_shape, root.dtype(), root.device());
561 }
562 *root_node->grad = grad_output;
563 }
564
565 std::cout << ">>> 根节点需要梯度，开始反向传播" << std::endl;
566 std::cout << ">>> 根节点形状: " << root_shape.size() << "D" << std::endl;
567
568 // 阶段3：反向传播（实现实际的梯度计算）
569 std::cout << ">>> 阶段3: 反向传播计算" << std::endl;
570
571 std::vector<Node*> topological_order;
572 {
573 std::lock_guard<std::mutex> lock(records_mutex);
574
575 // 生成拓扑序列
576 std::unordered_set<Node*> visited;
577 std::vector<Node*> forward_order;
578 dfs_topological_sort(root_node, visited, forward_order);
579
580 // 反转得到逆拓扑序列（反向传播顺序）
581 topological_order = forward_order;
582 std::reverse(topological_order.begin(), topological_order.end());
583
584 std::cout << ">>> 拓扑排序完成，节点数量: " << topological_order.size() << std::endl;
585 }
586
587 // 处理每个节点，传播梯度到输入节点
588 {
589 std::lock_guard<std::mutex> lock(records_mutex);
590 for (Node* node : topological_order) {
591 // 跳过不需要梯度的节点
592 if (!node->requires_grad) {
593 continue;
594 }
595
596 // 计算梯度并传播到输入节点
597 if (node->grad) {
598 // 对于标量情况，直接获取梯度值
599 float grad_val = node->grad->item<float>();
600
601 for (size_t input_id : node->input_ids) {
602 auto input_it = id_to_node.find(input_id);
603 if (input_it != id_to_node.end() && input_it->second && input_it->second->requires_grad) {
604 auto& input_node = *input_it->second;
605
606 // 初始化输入节点的梯度
607 if (!input_node.grad) {
608 input_node.grad = std::make_unique<Tensor>(ShapeTag{}, input_node.tensor->sizes(), input_node.tensor->dtype(), input_node.tensor->device());
609 input_node.grad->zero();
610 }
611
612 // 根据操作类型计算梯度
613 float input_grad_val = 0.0f;
614
615 // 处理不同操作类型
616 if (node->operation == op::Add) {
617 // ∂(a+b)/∂a = 1, ∂(a+b)/∂b = 1
618 input_grad_val = grad_val;
619 } else if (node->operation == op::Sub) {
620 // 第一个输入是a，第二个是b，∂(a-b)/∂a = 1, ∂(a-b)/∂b = -1
621 input_grad_val = grad_val * (input_id == node->input_ids[0] ? 1.0f : -1.0f);
622 } else if (node->operation == op::Mul) {
623 // 找到另一个输入张量
624 size_t other_input_id = (input_id == node->input_ids[0]) ? node->input_ids[1] : node->input_ids[0];
625 auto other_it = id_to_node.find(other_input_id);
626 if (other_it != id_to_node.end() && other_it->second) {
627 input_grad_val = grad_val * other_it->second->tensor->item<float>();
628 }
629 } else if (node->operation == op::Div) {
630 // 第一个输入是a，第二个是b，∂(a/b)/∂a = 1/b, ∂(a/b)/∂b = -a/(b^2)
631 if (input_id == node->input_ids[0]) {
632 // ∂(a/b)/∂a = 1/b
633 size_t b_id = node->input_ids[1];
634 auto b_it = id_to_node.find(b_id);
635 if (b_it != id_to_node.end() && b_it->second) {
636 float b_val = b_it->second->tensor->item<float>();
637 input_grad_val = grad_val / b_val;
638 }
639 } else {
640 // ∂(a/b)/∂b = -a/(b^2)
641 size_t a_id = node->input_ids[0];
642 auto a_it = id_to_node.find(a_id);
643 auto b_it = id_to_node.find(input_id);
644 if (a_it != id_to_node.end() && a_it->second && b_it != id_to_node.end() && b_it->second) {
645 float a_val = a_it->second->tensor->item<float>();
646 float b_val = b_it->second->tensor->item<float>();
647 input_grad_val = -grad_val * a_val / (b_val * b_val);
648 }
649 }
650 } else if (node->operation == op::Neg) {
651 // ∂(-a)/∂a = -1
652 input_grad_val = -grad_val;
653 } else if (node->operation == op::ReLU) {
654 // ∂ReLU(x)/∂x = 1 if x > 0 else 0
655 float x_val = input_node.tensor->item<float>();
656 input_grad_val = grad_val * (x_val > 0.0f ? 1.0f : 0.0f);
657 } else {
658 std::cout << "!!! 不支持的操作类型: " << static_cast<int>(node->operation) << std::endl;
659 }
660
661 // 累加梯度 - 直接访问梯度数据
662 float* grad_data = input_node.grad->data<float>();
663 if (grad_data) {
664 *grad_data += input_grad_val;
665 }
666 std::cout << ">>> 节点 " << node->tensor_id << " 向节点 " << input_id << " 传播梯度: " << input_grad_val << std::endl;
667 }
668 }
669 }
670 }
671 }
672
673 std::cout << ">>> 反向传播完成" << std::endl;
674 std::cout << "<<< =========================================" << std::endl;
675}
