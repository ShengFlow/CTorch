332

372// ======================= commit_record函数 =======================
373void AutoDiff::commit_record(Tensor &output) {
374 size_t output_id = output.id();
375 Ctorch_Error::trace(ErrorPlatform::kCPU,
376 "AutoDiff::commit_record - 开始, ID: " + std::to_string(output_id));
377
378 if (output_id == 0) {
379 Ctorch_Error::trace(ErrorPlatform::kCPU, "错误: 输出ID为0");
380 return;
381 }
382
383 debug_print_state("commit_record开始前");
384
385 // 第一步：收集必要信息（持有锁，但时间很短）
386 PendingRecord record_copy;
387 std::vector<size_t> input_ids_copy;
388 bool should_create_node = false;
389
390 {
391 std::lock_guard<std::mutex> lock(records_mutex);
392
393 auto it = pending_records.find(output_id);
394 if (it == pending_records.end()) {
395 Ctorch_Error::trace(ErrorPlatform::kCPU,
396 "警告: 找不到待处理记录 " + std::to_string(output_id));
397 return;
398 }
399
400 PendingRecord &record = it->second;
401 if (record.committed) {
402 Ctorch_Error::trace(ErrorPlatform::kCPU,
403 "记录 " + std::to_string(output_id) + " 已提交，跳过");
404 return;
405 }
406
407 Ctorch_Error::trace(ErrorPlatform::kCPU, "开始提交记录 " + std::to_string(output_id));
408 Ctorch_Error::trace(ErrorPlatform::kCPU,
409 "操作类型: " + std::to_string(static_cast<int>(record.operation)));
410 std::ostringstream oss;
411 oss << "输入IDs: ";
412 for (auto id : record.input_ids)
413 oss << id << " ";
414 Ctorch_Error::trace(ErrorPlatform::kCPU, oss.str());
415
416 // 验证输入节点
417 bool valid = true;
418 for (size_t input_id : record.input_ids) {
419 if (id_to_node.find(input_id) == id_to_node.end()) {
420 Ctorch_Error::trace(ErrorPlatform::kCPU,
421 "错误: 输入节点 " + std::to_string(input_id) + " 不存在");
422 valid = false;
423 break;
424 } else {
425 Ctorch_Error::trace(ErrorPlatform::kCPU,
426 "输入节点 " + std::to_string(input_id) + " 存在");
427 }
428 }
429
430 if (!valid) {
431 Ctorch_Error::trace(ErrorPlatform::kCPU,
432 "记录验证失败，清除记录 " + std::to_string(output_id));
433 pending_records.erase(it);
434 return;
435 }
436
437 // 复制必要信息，然后释放锁
438 record_copy = record;
439 input_ids_copy = record.input_ids;
440 should_create_node = true;
441
442 // 立即标记为已提交，避免重复处理
443 record.committed = true;
444 } // 释放锁
445
446 // 第二步：在锁外创建节点（避免死锁）
447 if (should_create_node) {
448 Ctorch_Error::trace(ErrorPlatform::kCPU, "在锁外创建操作节点 " + std::to_string(output_id));
449
450 // 确定梯度需求（在锁外检查）
451 bool requires_grad = false;
452 for (size_t input_id : input_ids_copy) {
453 Node *input_node = get_node_by_id(input_id);
454 if (input_node && input_node->requires_grad) {
455 requires_grad = true;
456 break;
457 }
458 }
459
460 Ctorch_Error::trace(ErrorPlatform::kCPU, "节点 " + std::to_string(output_id) +
461 " 需要梯度: " + (requires_grad ? "1" : "0"));
462
463 // 创建节点（不持有锁）
464 // 使用原始张量的拷贝，确保ID一致
465 auto tensor_ptr = std::make_unique<Tensor>(output);
466 auto output_node =
467 std::make_unique<Node>(output_id, std::move(tensor_ptr), requires_grad, false);
468 // 确保操作类型被正确设置
469 output_node->operation = record_copy.operation;
470 output_node->op_param_i = record_copy.op_param_i;
471 // 添加调试信息
472 std::ostringstream oss;
473 oss << "设置节点 " << output_id << " 的操作类型为: " << static_cast<int>(record_copy.operation);
474 Ctorch_Error::trace(ErrorPlatform::kCPU, oss.str());

476
477 // 延迟创建梯度存储
478 if (requires_grad) {
479 Ctorch_Error::trace(ErrorPlatform::kCPU,
480 "为节点 " + std::to_string(output_id) + " 延迟分配梯度存储");
481 // 注意：不在构造函数中创建 grad，避免死锁
482 }
483

485 {
486 std::lock_guard<std::mutex> lock(records_mutex);
487 id_to_node[output_id] = std::move(output_node);
488
489 // 清理已提交的记录
490 auto it = pending_records.find(output_id);
491 if (it != pending_records.end() && it->second.committed) {
492 pending_records.erase(it);
493 }
494 }
495
496 Ctorch_Error::trace(ErrorPlatform::kCPU, "记录提交完成 " + std::to_string(output_id));
497 debug_print_state("commit_record完成后");
498 }

1058void AutoDiff::dfs_topological_sort(Node *node, std::unordered_set<Node *> &visited,
1059 std::vector<Node *> &result) {
1060 if (!node)
1061 return;
1062 if (visited.find(node) != visited.end())
1063 return;
1064
1065 visited.insert(node);
1066
1067 // 先访问所有输入节点（依赖节点）
1068 for (size_t input_id : node->input_ids) {
1069 auto input_it = id_to_node.find(input_id);
1070 if (input_it != id_to_node.end() && input_it->second) {
1071 dfs_topological_sort(input_it->second.get(), visited, result);
1072 }
1073 }
1074
1075 // 后序遍历：当前节点处理完所有依赖后，加入结果
1076 result.push_back(node);
1077}

1079// ======================= backward函数（版本2） =======================
1080void AutoDiff::backward(Tensor &root, Tensor grad_output) {
1081 Ctorch_Error::trace(ErrorPlatform::kAutoDiff,
1082 std::string("======================================="));
1083 Ctorch_Error::trace(ErrorPlatform::kAutoDiff,
1084 std::string("进入 backward 函数，root ID: ") + std::to_string(root.id()));
1085

1093 Ctorch_Error::trace(ErrorPlatform::kAutoDiff, std::string("阶段1: 准备阶段"));
1094

1096 Ctorch_Error::trace(ErrorPlatform::kAutoDiff, std::string("阶段2: 验证和设置根节点梯度"));
1097 Node *root_node = nullptr;
1098 bool root_requires_grad = false;
1099 std::vector<size_t> root_shape;
1100
1101 {
1102 std::lock_guard<std::mutex> lock(records_mutex);
1103 auto it = id_to_node.find(root.id());
1104 if (it == id_to_node.end() || !it->second) {
1105 Ctorch_Error::trace(ErrorPlatform::kCPU,
1106 "错误: 找不到根节点 " + std::to_string(root.id()));
1107 return;

1111 root_shape = root.sizes();
1112
1113 if (!root_requires_grad) {
1114 Ctorch_Error::trace(ErrorPlatform::kCPU, "根节点不需要梯度，跳过反向传播");

1119 if (!root_node->grad) {
1120 root_node->grad =

1122 }
1123
1124 // 检查并调整梯度输出形状，确保与根节点形状匹配

1126 *root_node->grad = adjusted_grad_output;
1127 }

1130 Ctorch_Error::trace(ErrorPlatform::kCPU,

1132 Ctorch_Error::trace(ErrorPlatform::kCPU,

1137

1139 {

1141
1142 // 生成拓扑序列

1158 for (Node *node : topological_order) {
1159 // 打印节点信息

1163 Ctorch_Error::trace(ErrorPlatform::kCPU,

1168 "D");

1173 }

1175 }
1176 }
1177
1178 // 跳过不需要梯度的节点
1179 if (!node->requires_grad) {
1180 Ctorch_Error::trace(ErrorPlatform::kCPU, "跳过不需要梯度的节点");

1185 if (node->grad) {
1186 // 对于张量梯度，直接使用梯度张量进行计算
1187 Tensor &node_grad = *node->grad;

1190 auto input_it = id_to_node.find(input_id);
1191 if (input_it != id_to_node.end() && input_it->second &&
1192 input_it->second->requires_grad) {
1193 auto &input_node = *input_it->second;
1194
1195 // 打印输入节点信息
1196 Ctorch_Error::trace(
1197 ErrorPlatform::kCPU,

1200

1202 if (!input_node.grad) {

1204 ShapeTag{}, input_node.tensor->sizes(), input_node.tensor->dtype(),
1205 input_node.tensor->device());

1207 Ctorch_Error::trace(ErrorPlatform::kCPU, "初始化输入节点梯度");
1208 }

1211
1212 // 根据操作类型计算梯度

1214
1215 // 处理不同操作类型
1216 if (node->operation == op::Add) {
1217 // ∂(a+b)/∂a = 1, ∂(a+b)/∂b = 1
1218 // 对于张量加法，梯度直接传播
1219 // 但需要确保梯度形状与输入张量形状匹配
1220 input_grad_val =
1221 check_and_adjust_grad_shape(node_grad, input_node.tensor->sizes());
1222 } else if (node->operation == op::Sub) {
1223 // 第一个输入是a，第二个是b，∂(a-b)/∂a = 1, ∂(a-b)/∂b = -1
1224 if (input_id == node->input_ids[0]) {
1225 // 对第一个输入的梯度是正的
1226 input_grad_val = check_and_adjust_grad_shape(
1227 node_grad, input_node.tensor->sizes());
1228 } else {
1229 // 对第二个输入的梯度是负的
1230 Tensor adjusted_grad = check_and_adjust_grad_shape(
1231 node_grad, input_node.tensor->sizes());
1232 input_grad_val = -adjusted_grad;
1233 }
1234 } else if (node->operation == op::Mul) {
1235 // 找到另一个输入张量
1236 if (node->input_ids.size() > 1) {
1237 size_t other_input_id = (input_id == node->input_ids[0])
1238 ? node->input_ids[1]
1239 : node->input_ids[0];
1240 auto other_it = id_to_node.find(other_input_id);
1241 if (other_it != id_to_node.end() && other_it->second) {
1242 // 对于张量乘法，梯度是另一个输入张量与当前节点梯度的乘积
1243 Tensor &other_tensor = *other_it->second->tensor;
1244 // 检查是否是同一个张量相乘
1245 if (input_id == other_input_id) {
1246 // 对于同一个张量相乘，梯度是 2 * other_tensor * node_grad
1247 // 但由于会遍历两次输入节点，这里只计算一次
1248 input_grad_val = other_tensor * node_grad;
1249 } else {
1250 // 对于不同张量相乘，梯度是 other_tensor * node_grad
1251 input_grad_val = other_tensor * node_grad;
1252 }
1253 }
1254 } else {
1255 // 对于标量乘法，梯度是标量值与当前节点梯度的乘积
1256 // 注意：这里我们假设标量值已经被应用到了结果中，所以梯度计算比较简单
1257 input_grad_val = node_grad;
1258 }
1259 } else if (node->operation == op::Div) {
1260 // 第一个输入是a，第二个是b，∂(a/b)/∂a = 1/b, ∂(a/b)/∂b = -a/(b^2)
1261 if (input_id == node->input_ids[0]) {
1262 // ∂(a/b)/∂a = 1/b
1263 size_t b_id = node->input_ids[1];
1264 auto b_it = id_to_node.find(b_id);
1265 if (b_it != id_to_node.end() && b_it->second) {
1266 Tensor &b_tensor = *b_it->second->tensor;
1267 // 检查b_tensor是否包含零值
1268 bool has_zero = false;
1269 size_t numel = b_tensor.numel();
1270 if (b_tensor.dtype() == DType::kFloat) {
1271 const float *b_data = b_tensor.data<float>();
1272 for (size_t i = 0; i < numel; ++i) {
1273 if (std::abs(b_data[i]) <
1274 std::numeric_limits<float>::epsilon()) {
1275 has_zero = true;
1276 break;
1277 }
1278 }
1279 }
1280 if (has_zero) {
1281 Ctorch_Error::throwException(
1282 ErrorPlatform::kGENERAL, ErrorType::TENSOR_STATE,
1283 "除零错误：Div操作的梯度计算中除数为零");
1284 }
1285 input_grad_val = node_grad / b_tensor;
1286 }
1287 } else {
1288 // ∂(a/b)/∂b = -a/(b^2)
1289 size_t a_id = node->input_ids[0];
1290 auto a_it = id_to_node.find(a_id);
1291 auto b_it = id_to_node.find(input_id);
1292 if (a_it != id_to_node.end() && a_it->second &&
1293 b_it != id_to_node.end() && b_it->second) {
1294 Tensor &a_tensor = *a_it->second->tensor;
1295 Tensor &b_tensor = *b_it->second->tensor;
1296 // 检查b_tensor是否包含零值
1297 bool has_zero = false;
1298 size_t numel = b_tensor.numel();
1299 if (b_tensor.dtype() == DType::kFloat) {
1300 const float *b_data = b_tensor.data<float>();
1301 for (size_t i = 0; i < numel; ++i) {
1302 if (std::abs(b_data[i]) <
1303 std::numeric_limits<float>::epsilon()) {
1304 has_zero = true;
1305 break;
1306 }
1307 }
1308 }
1309 if (has_zero) {
1310 Ctorch_Error::throwException(
1311 ErrorPlatform::kGENERAL, ErrorType::TENSOR_STATE,
1312 "除零错误：Div操作的梯度计算中除数为零");
1313 }
1314 input_grad_val = -a_tensor * node_grad / (b_tensor * b_tensor);
1315 }
1316 }
1317 } else if (node->operation == op::Neg) {
1318 // ∂(-a)/∂a = -1
1319 input_grad_val = -node_grad;
1320 } else if (node->operation == op::ReLU) {
1321 // ∂ReLU(x)/∂x = 1 if x > 0 else 0
1322 // 对于张量ReLU，梯度是node_grad乘以x>0的掩码
1323 Tensor &x_tensor = *input_node.tensor;
1324
1325 // 实现ReLU梯度计算：直接在梯度上乘以mask
1326 // 这里创建一个与x_tensor形状相同的张量，x>0时为1，否则为0
1327 Tensor mask(ShapeTag{}, x_tensor.sizes(), x_tensor.dtype(),
1328 x_tensor.device());
1329 size_t numel = x_tensor.numel();
1330
1331 // 根据数据类型处理
1332 switch (x_tensor.dtype()) {
1333 case DType::kFloat: {
1334 float *mask_data = mask.data<float>();
1335 const float *x_data = x_tensor.data<float>();
1336 for (size_t i = 0; i < numel; ++i) {
1337 mask_data[i] = (x_data[i] > 0.0f) ? 1.0f : 0.0f;
1338 }
1339 break;
1340 }
1341 case DType::kDouble: {
1342 double *mask_data = mask.data<double>();
1343 const double *x_data = x_tensor.data<double>();
1344 for (size_t i = 0; i < numel; ++i) {
1345 mask_data[i] = (x_data[i] > 0.0) ? 1.0 : 0.0;
1346 }
1347 break;
1348 }
1349 case DType::kInt: {
1350 int32_t *mask_data = mask.data<int32_t>();
1351 const int32_t *x_data = x_tensor.data<int32_t>();
1352 for (size_t i = 0; i < numel; ++i) {
1353 mask_data[i] = (x_data[i] > 0) ? 1 : 0;
1354 }
1355 break;
1356 }
1357 case DType::kLong: {
1358 int64_t *mask_data = mask.data<int64_t>();
1359 const int64_t *x_data = x_tensor.data<int64_t>();
1360 for (size_t i = 0; i < numel; ++i) {
1361 mask_data[i] = (x_data[i] > 0) ? 1 : 0;
1362 }
1363 break;
1364 }
1365 default:
1366 Ctorch_Error::throwException(ErrorPlatform::kCPU,
1367 ErrorType::DATATYPE,
1368 "ReLU梯度计算中不支持的dtype");
1369 }
1370
1371 input_grad_val = node_grad * mask;
1372 } else if (node->operation == op::Sigmoid) {
1373 // ∂sigmoid(x)/∂x = sigmoid(x) * (1 - sigmoid(x)) * grad
1374 Tensor &x_tensor = *input_node.tensor;
1375
1376 // 计算 sigmoid(x)
1377 Tensor sigmoid_x = x_tensor.sigmoid();
1378
1379 // 计算 1 - sigmoid(x)
1380 Tensor one_minus_sigmoid_x = Tensor(1.0f) - sigmoid_x;
1381
1382 // 计算梯度
1383 input_grad_val = sigmoid_x * one_minus_sigmoid_x * node_grad;
1384 } else if (node->operation == op::Tanh) {
1385 // ∂tanh(x)/∂x = (1 - tanh(x)^2) * grad
1386 Tensor &x_tensor = *input_node.tensor;
1387
1388 // 计算 tanh(x)
1389 Tensor tanh_x = x_tensor.tanh();
1390
1391 // 计算 tanh(x)^2
1392 Tensor tanh_x_squared = tanh_x * tanh_x;
1393
1394 // 计算 1 - tanh(x)^2
1395 Tensor one_minus_tanh_x_squared = Tensor(1.0f) - tanh_x_squared;
1396
1397 // 计算梯度
1398 input_grad_val = one_minus_tanh_x_squared * node_grad;
1399 } else if (node->operation == op::Sin) {
1400 // ∂sin(x)/∂x = cos(x) * grad
1401 Tensor &x_tensor = *input_node.tensor;
1402
1403 // 计算 cos(x)
1404 Tensor cos_x = x_tensor.cos();
1405
1406 // 计算梯度
1407 input_grad_val = cos_x * node_grad;
1408 } else if (node->operation == op::Cos) {
1409 // ∂cos(x)/∂x = -sin(x) * grad
1410 Tensor &x_tensor = *input_node.tensor;
1411
1412 // 计算 sin(x)
1413 Tensor sin_x = x_tensor.sin();
1414
1415 // 计算梯度
1416 input_grad_val = -sin_x * node_grad;
1417 } else if (node->operation == op::Softmax) {
1418 // 支持 dim 的 softmax 反向：dL/dx = y * (g - sum(g*y, axis=dim))
1419 // 这里使用 node->tensor（softmax 输出 y）而不是重新计算 softmax(x)
1420 // node->op_param_i 保存 dim（由 Tensor::softmax(dim) 记录）
1421 int dim = node->op_param_i;
1422 const Tensor &y_tensor = *node->tensor; // softmax 输出
1423 const Tensor &g_tensor = node_grad; // dL/dy
1424
1425 std::vector<size_t> shape = y_tensor.sizes();
1426 if (y_tensor.dtype() != DType::kFloat) {
1427 Ctorch_Error::throwException(ErrorPlatform::kCPU, ErrorType::DATATYPE,
1428 "Softmax backward: 目前仅支持 float");
1429 }
1430 if (shape.size() == 1) {
1431 // 1D: dim 只允许 0 或 -1
1432 if (dim == -1) dim = 0;
1433 if (dim != 0) {
1434 Ctorch_Error::throwException(ErrorPlatform::kCPU, ErrorType::DIMENSION,
1435 "Softmax backward: 1D 张量仅支持 dim=0/-1");
1436 }
1437 Tensor dx(ShapeTag{}, shape, y_tensor.dtype(), y_tensor.device());
1438 size_t n = y_tensor.numel();
1439 const float* y = y_tensor.data<float>();
1440 const float* g = g_tensor.data<float>();
1441 float* out = dx.data<float>();
1442
1443 float dot = 0.0f;
1444 for (size_t i = 0; i < n; ++i) dot += g[i] * y[i];
1445 for (size_t i = 0; i < n; ++i) out[i] = y[i] * (g[i] - dot);
1446 input_grad_val = dx;
1447 } else if (shape.size() == 2) {
1448 // 2D: 支持 dim=0 或 dim=1 或 -1(默认最后一维=1)
1449 if (dim == -1) dim = 1;
1450 if (dim != 0 && dim != 1) {
1451 Ctorch_Error::throwException(ErrorPlatform::kCPU, ErrorType::DIMENSION,
1452 "Softmax backward: 2D 张量仅支持 dim=0/1/-1");
1453 }
1454 size_t rows = shape[0];
1455 size_t cols = shape[1];
1456 Tensor dx(ShapeTag{}, shape, y_tensor.dtype(), y_tensor.device());
1457 const float* y = y_tensor.data<float>();
1458 const float* g = g_tensor.data<float>();
1459 float* out = dx.data<float>();
1460
1461 if (dim == 1) {
1462 // 按行
1463 for (size_t i = 0; i < rows; ++i) {
1464 float dot = 0.0f;
1465 for (size_t j = 0; j < cols; ++j) {
1466 dot += g[i * cols + j] * y[i * cols + j];
1467 }
1468 for (size_t j = 0; j < cols; ++j) {
1469 size_t idx = i * cols + j;
1470 out[idx] = y[idx] * (g[idx] - dot);
1471 }
1472 }
1473 } else {
1474 // 按列
1475 for (size_t j = 0; j < cols; ++j) {
1476 float dot = 0.0f;
1477 for (size_t i = 0; i < rows; ++i) {
1478 dot += g[i * cols + j] * y[i * cols + j];
1479 }
1480 for (size_t i = 0; i < rows; ++i) {
1481 size_t idx = i * cols + j;
1482 out[idx] = y[idx] * (g[idx] - dot);
1483 }
1484 }
1485 }
1486 input_grad_val = dx;
1487 } else {
1488 Ctorch_Error::throwException(ErrorPlatform::kCPU, ErrorType::DIMENSION,
1489 "Softmax backward: 暂仅支持 1D/2D 张量");
1490 }
1491 } else if (node->operation == op::MSE) {
1492 // ∂MSE(pred, target)/∂pred = 2 * (pred - target) / n * grad
1493 // 找到另一个输入张量（目标张量）
1494 size_t target_id = (input_id == node->input_ids[0])
1495 ? node->input_ids[1]
1496 : node->input_ids[0];
1497 auto target_it = id_to_node.find(target_id);
1498 if (target_it != id_to_node.end() && target_it->second) {
1499 Tensor &pred_tensor = *input_node.tensor;
1500 Tensor &target_tensor = *target_it->second->tensor;
1501
1502 // 计算 pred - target
1503 Tensor pred_minus_target = pred_tensor - target_tensor;
1504
1505 // 计算 2 * (pred - target)
1506 Tensor two_times_diff = pred_minus_target * 2.0f;
1507
1508 // 计算元素数量
1509 size_t n = pred_tensor.numel();
1510
1511 // 计算 2 * (pred - target) / n
1512 Tensor grad_contrib = two_times_diff / static_cast<float>(n);
1513
1514 // 计算梯度
1515 input_grad_val = grad_contrib * node_grad;
1516 }
1517 } else if (node->operation == op::CE) {
1518 // ∂CrossEntropy(pred, target)/∂pred = (softmax(pred) - target) * grad
1519 // 找到另一个输入张量（目标张量）
1520 size_t target_id = (input_id == node->input_ids[0])
1521 ? node->input_ids[1]
1522 : node->input_ids[0];
1523 auto target_it = id_to_node.find(target_id);
1524 if (target_it != id_to_node.end() && target_it->second) {
1525 Tensor &pred_tensor = *input_node.tensor;
1526 Tensor &target_tensor = *target_it->second->tensor;
1527
1528 std::vector<size_t> pred_shape = pred_tensor.sizes();
1529 if (pred_tensor.dtype() != DType::kFloat) {
1530 Ctorch_Error::throwException(ErrorPlatform::kCPU, ErrorType::DATATYPE,
1531 "CrossEntropy backward: 目前仅支持 float");
1532 }
1533
1534 // 只处理 logits 为 1D/2D 的常见分类场景：
1535 // - 1D: [num_classes]（单样本）
1536 // - 2D: [batch_size, num_classes]
1537 if (pred_shape.size() == 2) {
1538 size_t batch_size = pred_shape[0];
1539 size_t num_classes = pred_shape[1];
1540 Tensor softmax_pred(ShapeTag{}, pred_shape, pred_tensor.dtype(), pred_tensor.device());
1541
1542 const float* pred_data = pred_tensor.data<float>();
1543 float* softmax_data = softmax_pred.data<float>();
1544
1545 // 对每一行（每个样本）计算 softmax
1546 for (size_t i = 0; i < batch_size; ++i) {
1547 // 找到该行的最大值（数值稳定性）
1548 float max_val = pred_data[i * num_classes];
1549 for (size_t j = 1; j < num_classes; ++j) {
1550 if (pred_data[i * num_classes + j] > max_val) {
1551 max_val = pred_data[i * num_classes + j];
1552 }
1553 }
1554
1555 // 计算 exp 和
1556 float exp_sum = 0.0f;
1557 for (size_t j = 0; j < num_classes; ++j) {
1558 exp_sum += std::exp(pred_data[i * num_classes + j] - max_val);
1559 }
1560
1561 // 计算 softmax
1562 for (size_t j = 0; j < num_classes; ++j) {
1563 softmax_data[i * num_classes + j] =
1564 std::exp(pred_data[i * num_classes + j] - max_val) / exp_sum;
1565 }
1566 }
1567
1568 // 计算 softmax(pred) - target
1569 Tensor softmax_minus_target = softmax_pred - target_tensor;
1570
1571 // 计算梯度：∂CE/∂pred = (softmax(pred) - target) * grad
1572 // grad 是标量（CE loss 是标量），需要广播
1573 float grad_val = node_grad.item<float>();
1574 input_grad_val = softmax_minus_target * grad_val;
1575 } else if (pred_shape.size() == 1) {
1576 // 1D logits：单样本
1577 size_t num_classes = pred_shape[0];
1578 Tensor softmax_pred(ShapeTag{}, pred_shape, pred_tensor.dtype(), pred_tensor.device());
1579 const float* pred_data = pred_tensor.data<float>();
1580 float* softmax_data = softmax_pred.data<float>();
1581
1582 float max_val = pred_data[0];
1583 for (size_t j = 1; j < num_classes; ++j) {
1584 if (pred_data[j] > max_val) max_val = pred_data[j];
1585 }
1586 float exp_sum = 0.0f;
1587 for (size_t j = 0; j < num_classes; ++j) {
1588 exp_sum += std::exp(pred_data[j] - max_val);
1589 }
1590 for (size_t j = 0; j < num_classes; ++j) {
1591 softmax_data[j] = std::exp(pred_data[j] - max_val) / exp_sum;
1592 }
1593
1594 Tensor softmax_minus_target = softmax_pred - target_tensor;
1595 float grad_val = node_grad.item<float>();
1596 input_grad_val = softmax_minus_target * grad_val;
1597 } else {
1598 Ctorch_Error::throwException(ErrorPlatform::kCPU, ErrorType::DIMENSION,
1599 "CrossEntropy backward: 暂仅支持 1D/2D logits");
1600 }
1601 }
1602 } else if (node->operation == op::MAE) {
1603 // ∂MAE(pred, target)/∂pred = sign(pred - target) / n * grad
1604 // 找到另一个输入张量（目标张量）
1605 size_t target_id = (input_id == node->input_ids[0])
1606 ? node->input_ids[1]
1607 : node->input_ids[0];
1608 auto target_it = id_to_node.find(target_id);
1609 if (target_it != id_to_node.end() && target_it->second) {
1610 Tensor &pred_tensor = *input_node.tensor;
1611 Tensor &target_tensor = *target_it->second->tensor;
1612
1613 // 计算 pred - target
1614 Tensor pred_minus_target = pred_tensor - target_tensor;
1615
1616 // 计算 sign(pred - target)
1617 Tensor sign_tensor(ShapeTag{}, pred_minus_target.sizes(),
1618 pred_minus_target.dtype(),
1619 pred_minus_target.device());
1620 float *sign_data = sign_tensor.data<float>();
1621 const float *diff_data = pred_minus_target.data<float>();
1622 size_t numel = pred_minus_target.numel();
1623
1624 for (size_t i = 0; i < numel; ++i) {
1625 if (diff_data[i] > 0.0f) {
1626 sign_data[i] = 1.0f;
1627 } else if (diff_data[i] < 0.0f) {
1628 sign_data[i] = -1.0f;
1629 } else {
1630 sign_data[i] = 0.0f;
1631 }
1632 }
1633
1634 // 计算元素数量
1635 size_t n = pred_tensor.numel();
1636
1637 // 计算 sign(pred - target) / n
1638 Tensor grad_contrib = sign_tensor / static_cast<float>(n);
1639
1640 // 计算梯度
1641 input_grad_val = grad_contrib * node_grad;
1642 }
1643 } else if (node->operation == op::MatMul) {
1644 // ∂(A*B)/∂A = grad * B^T, ∂(A*B)/∂B = A^T * grad
1645 // 找到另一个输入张量
1646 size_t other_input_id = (input_id == node->input_ids[0])
1647 ? node->input_ids[1]
1648 : node->input_ids[0];
1649 auto other_it = id_to_node.find(other_input_id);
1650 if (other_it != id_to_node.end() && other_it->second) {
1651 Tensor &other_tensor = *other_it->second->tensor;
1652 if (input_id == node->input_ids[0]) {
1653 // 对第一个输入的梯度是 grad * B^T
1654 try {
1655 // 打印调试信息
1656 Ctorch_Error::trace(ErrorPlatform::kCPU,
1657 "MatMul梯度计算：处理第一个输入");
1658 Ctorch_Error::trace(
1659 ErrorPlatform::kCPU,
1660 "输入形状: " +
1661 std::to_string(input_node.tensor->sizes()[0]) +
1662 "x" +
1663 std::to_string(input_node.tensor->sizes()[1]));
1664 Ctorch_Error::trace(
1665 ErrorPlatform::kCPU,
1666 "另一个输入形状: " +
1667 std::to_string(other_tensor.sizes()[0]) + "x" +
1668 std::to_string(other_tensor.sizes()[1]));
1669 Ctorch_Error::trace(
1670 ErrorPlatform::kCPU,
1671 "节点梯度形状: " +
1672 std::to_string(node_grad.sizes().size()) + "D");
1673 if (!node_grad.sizes().empty()) {
1674 Ctorch_Error::trace(
1675 ErrorPlatform::kCPU,
1676 "节点梯度维度: " +
1677 std::to_string(node_grad.sizes()[0]) + "x" +
1678 std::to_string(node_grad.sizes()[1]));
1679 }
1680
1681 // 获取 node_grad 的形状和值
1682 bool is_scalar_grad = node_grad.sizes().empty();
1683
1684 if (is_scalar_grad) {
1685 // 标量情况：grad * B^T 等同于 B^T * grad_value
1686 float grad_value = node_grad.item<float>();
1687 Ctorch_Error::trace(ErrorPlatform::kCPU,
1688 "标量梯度值: " +
1689 std::to_string(grad_value));
1690
1691 // 计算 B^T 并乘以 grad_value
1692 Tensor B_T = other_tensor.t();
1693 input_grad_val = B_T * grad_value;
1694 } else {
1695 // 张量情况：grad * B^T
1696 if (node_grad.sizes().size() != 2) {
1697 throw std::runtime_error(
1698 "MatMul gradient must be 2D tensor or scalar");
1699 }
1700
1701 // 计算 B^T
1702 Tensor B_T = other_tensor.t();
1703 // 使用现有的matmul函数计算 grad * B^T
1704 input_grad_val = node_grad.matmul(B_T);
1705 }
1706
1707 // 验证计算结果是否为非零张量
1708 bool result_is_zero = true;
1709 if (!input_grad_val.sizes().empty()) {
1710 size_t numel = input_grad_val.numel();
